import requests
import json

base_url = "https://cmsti.stj.jus.br/"

headers = {"Content-Type": "application/x-www-form-urlencoded"}

body = {
    "grant_type": "password",
    "client_id": "front-manager",
    "username": "adm.cmsti",
    "password": "prd_prod",
}

response = requests.post(f"{base_url}/auth/realms/stj-prd/protocol/openid-connect/token", headers=headers, data=body)

auth_token = response.json()["access_token"]

print(f"Authentication token: {auth_token}")
