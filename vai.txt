import requests
import json

base_url = "https://cmsti.stj.jus.br/"
auth_token = "YOUR_AUTH_TOKEN"
cit_user = "adm.cmsti"
cit_pass = "prd_prod"
client_id = "front-manager"

headers = {"Content-Type": "application/x-www-form-urlencoded"}

    "client_id": client_id,
    "username": cit_user,
